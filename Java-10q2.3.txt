import student.StudentInfo;
import student.StudentPer;
import java.util.Scanner;
public class Main{
public static void main(String[] args){
Scanner sc=new Scanner(System.in);
System.out.print("Enter Roll No: ");
int rollNo=sc.nextInt();
sc.nextLine();
System.out.print("Enter Name: ");
String name=sc.nextLine();
System.out.print("Enter Class: ");
String studentClass=sc.nextLine();
int[] marks=new int[6];
System.out.println("Enter marks of 6 subjects (out of 100):");
for(int i=0;i<6;i++){
System.out.print("Subject "+(i+1)+": ");
marks[i]=sc.nextInt();
}
StudentPer studentPer=new StudentPer();
double percentage=studentPer.calculatePercentage(marks);
StudentInfo studentInfo=new StudentInfo(rollNo,name,studentClass,percentage);
studentInfo.displayInfo();
sc.close();
}
}
