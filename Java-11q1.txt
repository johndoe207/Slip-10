interface Operation {
    double PI = 3.142;
    void volume();     
}
class Cylinder implements Operation {
    double radius;
    double height;
        Cylinder(double radius, double height) {
        this.radius = radius;
        this.height = height;
    }
       public void volume() {
        double vol = PI * radius * radius * height;
        System.out.println("Volume of Cylinder = " + vol);
    }
}
public class Main {
    public static void main(String[] args) {
        // Create object of Cylinder
        Cylinder c = new Cylinder(3.0, 7.0);  // Example values
        c.volume(); // Call method to calculate and print volume
    }
}
